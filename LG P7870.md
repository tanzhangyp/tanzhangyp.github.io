## 题解：洛谷 P7870 「Wdoi-4」兔已着陆

> 题目地址：[P7870 「Wdoi-4」兔已着陆 - 洛谷 | 计算机科学教育新生态 (luogu.com.cn)](https://www.luogu.com.cn/problem/P7870)

### 20分解法：

```cpp
// P7870 「Wdoi-4」兔已着陆
// 第2个测试点，对于另外 20% 的数据，l=r
#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int n, f, l, r, zhan[500005], t = 0, j;
    long long k;
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        long long sum = 0;
        cin >> f;
        if (f == 1)
        {
            cin >> l >> r;
            // 按照题意如果是1，依次入栈
            // 由于l=r，选一个入就可以了
            zhan[++t] = l;
        }
        else
        {
            cin >> k;
            // 由于每轮只入了1个，按照现有的栈个数往回出k个
            for (j = t; j > t - k; j--)
                // 累加每次出栈的数字
                sum += zhan[j];
            // 重新确定最右面的栈
            t = j;
            cout << sum << endl;
        }
    }
    return 0;
}
```



### 50分解法：

```cpp
// P7870 「Wdoi-4」兔已着陆
// O(n2)复杂度，n的范围为10^5，只能得50分
#include <iostream>
#include <cstdio>
#include <stack>
using namespace std;
// 定义栈
stack<int> s;
int main()
{
	// freopen在新版本的编译器中已经集成在iostream
	// 为了保险起见，还是要加上cstdio
	//	freopen("stack2.in","r",stdin);
	//	freopen("stack2.out","w",stdout);
	// f为判断是入栈1还是出站2
	int n, f, l, r; 
	long long k;
	cin >> n;
	for (int i = 1; i <= n; i++)
	{
		long long sum = 0;
		cin >> f;
		if (f == 1)
		{
			cin >> l >> r;
			// 按照题意如果是1，依次入栈
			for (int j = l; j <= r; j++)
				s.push(j);
		}
		else
		{
			cin >> k;
			// 按照k的个数依次出栈
			for (int p = 1; p <= k; p++)
			{
				// 累加每次出栈的数字
				sum += s.top();
				// 出栈
				s.pop();
			}
			cout << sum << endl;
		}
	}
	return 0;
}
```



### 100分解法：

```cpp
// P7870 「Wdoi-4」兔已着陆
// O(n2)通过判断和整段操作大幅度优化
#include <iostream>
#include <cstdio>
using namespace std;

// 定义模拟入栈的结构体,n的范围是10^5
// 实际入栈的肯定少于100005
struct zhan
{
	// 每次入栈的最左面和最右面的数字
	// 测试点最大为10^6，int足以
	int l, r; //long long l,r;
	// len为模拟入栈数字的长度，按照题意入栈数字最大为10^6，实际上int就够了，
	// 但是下面其参与了和lr的高斯运算，所以应为ll
	// 更好的方法是为了保险，包括lr在内都为ll
	long long len;
} a[500005]; //n为5*10^5，实际入栈的用不了这么多

int main()
{
	//	freopen("stack2.in","r",stdin);
	//	freopen("stack2.out","w",stdout);
	// f为判断是入栈1还是出站2
	// j为实际存入栈的计数变量
	int n, f, j = 1;
	// k为10^12，longlong为19位
	long long k;
	cin >> n;
	// i是所有入栈和出站的编号
	for (int i = 1; i <= n; i++)
	{
		cin >> f;
		// 入栈操作
		if (f == 1)
		{
			// j只负责入栈
			cin >> a[j].l >> a[j].r;
			// 根据最左面和最右面计算出长度
			a[j].len = a[j].r - a[j].l + 1;
			j++;
		}
		// 出站操作
		else
		{
			cin >> k;
			// j不能动，用临时变量t
			int t = j;
			// k最大为12位，2*500000500000，累加和最大为13位，19位足以
			long long ans = 0;
			// k还没有出栈完 而且 临时t从前往后还没有消耗完
			while (k > 0 && t--)
			{
				// 如果k的长度在栈顶所处的段中而且k没有出完
				if (k <= a[t].len && k != 0)
				{
					// 高斯，因为是连续数，计算右数到k的范围
					ans += (a[t].r - k + 1 + a[t].r) * k / 2;
					// 更新右数
					a[t].r -= k;
					// 更新长度
					a[t].len -= k;
					// k用完了
					k = 0;
				}
				// 如果k超过了当前的栈顶所在段而且当前的段
				else if (k > a[t].len && a[t].len > 0)
				{
					// 计算整个当前段
					ans += (a[t].l + a[t].r) * a[t].len / 2;
					// k还有多少没算
					k -= a[t].len;
					// 当前段清零
					a[t].len = 0;
				}
			}
			cout << ans << endl;
		}
	}
	return 0;
}
```
