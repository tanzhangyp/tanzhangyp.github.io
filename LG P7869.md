# 洛谷 P7869 「Wdoi-4」使用三个系统程度的能力

### 使用字符串解法

```cpp
//P7869 「Wdoi-4」使用三个系统程度的能力
//使用朴素的字符串下标进行判断
#include <iostream>
#include <string>
using namespace std;
int main()
{
    string s;
    // 读取整行忽略空格
    getline(cin, s);
    // 取得字符串长度
    int len = s.size();
    for (int i = 0; i < len; i++)
    {
        // 碰见的第一个\，注意\是转义序列符号，前面还需要加一个\，才能当做字符处理
        if (s[i] == '\\')
        {
            // \下一个字符是r
            if (s[i + 1] == 'r')
            {
                // \后第2个字符是\，第3个是n
                if (s[i + 2] == '\\' && s[i + 3] == 'n')
                {
                    cout << "windows";
                    //注意找到后就结束循环，因题目说明这个文本文件是在单一系统中写成的，因此保证换行符只会出现上述三种情况之一
                    //本身真是操作系统也是如此
                    break;
                }
                // 依次排除
                cout << "mac";
                break;
            }
            else
            {
                cout << "linux";
                break;
            }
        }
    }
    return 0;
}
```



### 使用字符查找函数

```cpp
//P7869 「Wdoi-4」使用三个系统程度的能力
//使用string的find函数
#include <iostream>
#include <string>
using namespace std;
int main()
{
    string s;
    // 读取整行忽略空格
    getline(cin, s);
    // s.find(str, [pos]):在字符串s中从第pos个字符开始寻找str，并返回位置，如果找不到返回-1。
    // pos可以省略，默认值是0。
    // "\\r",注意要查找的字符串中的\也要多加一个
    // 利用函数查找，不再需要循环，但是查找每次都是从第0个字符开始，按道理说没有普通循环那种挨个排除快
    int flag = s.find("\\r");
    if (flag != -1)
    {
        flag = s.find("\\n", 0);
        if (flag != -1)
            cout << "windows";
        else
            cout << "mac";
    }
    else
        cout << "linux";
    return 0;
}
```
